//  判断是苹果还是window电脑
function option_os() {
  var agent = navigator.userAgent.toLowerCase();
  var isMac = /macintosh|mac os x/i.test(navigator.userAgent);
  if (isMac) {
    return "mac";
  } else if (agent.indexOf("win64") >= 0 || agent.indexOf("wow64") >= 0) {
    return "win64";
  } else if (agent.indexOf("win32") >= 0 || agent.indexOf("wow32") >= 0) {
    return "win64";
  }
}
